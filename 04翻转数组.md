import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int m = sc.nextInt();
        if(n<2||n>1000000000||m<1||n%(2*m)!=0){
            System.out.print(0);
            return;
        }
        long sum = n/(2*m)*m;
        sum*=m;
        System.out.println(sum);
    }
}
